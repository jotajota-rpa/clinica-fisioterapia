==============================
DOCKER - COMANDOS MÁS USADOS
==============================

# Construir y levantar tu proyecto
docker-compose up -d --build  # Levantar en segundo plano ygit construir
docker-compose logs -f        # Ver logs de todos los contenedores
docker-compose down           # Parar todos los contenedores

# Ver estado y uso de recursos
docker ps                     # Contenedores activos
docker ps -a                  # Todos los contenedores
docker stats                  # Uso de CPU/RAM de contenedores
docker system df              # Espacio ocupado por Docker

# Limpiar espacio y basura sin tocar contenedores activos
docker container prune        # Borrar contenedores parados
docker image prune            # Borrar imágenes no usadas
docker network prune          # Borrar redes no usadas
docker system prune           # Limpieza general ligera

# Limpieza más agresiva (cuidado)
docker system prune -a --volumes   # Borra todo lo no usado, incl. volúmenes
docker-compose down -v --rmi all   # Borra todo

# Reconstruir contenedor frontend
docker-compose stop frontend
docker-compose rm -f frontend
docker-compose up --build -d frontend
docker-compose logs -f frontend

# Instalar paquetes en npm
docker-compose run --rm frontend npm install
